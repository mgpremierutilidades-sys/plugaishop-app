#!/bin/sh
# Plugaishop: guardrails antes de commitar
# - typecheck/lint
# - verificações anti-"opacidade"

set -e

# Windows/PowerShell friendly
if command -v pwsh >/dev/null 2>&1; then
  pwsh -NoProfile -ExecutionPolicy Bypass -File "scripts/hooks/run-checks.ps1" -Mode "pre-commit"
else
  powershell -NoProfile -ExecutionPolicy Bypass -File "scripts/hooks/run-checks.ps1" -Mode "pre-commit"
fi
